        + generateNextId(): int

package ru.codehub.command {
    interface Command {
        + getName(): String
        + getDescription(): String
        + execute(args: String[], scanner: java.util.Scanner): void
    }

    class CommandRegistry {
        - commands: Map<String, Command>
        - history: List<String>
        + CommandRegistry()
        + register(command: Command): void
        + getCommand(name: String): Command
        + getCommands(): Collection<Command>
        + addToHistory(commandName: String): void
        + getHistory(): List<String>
    }
}
